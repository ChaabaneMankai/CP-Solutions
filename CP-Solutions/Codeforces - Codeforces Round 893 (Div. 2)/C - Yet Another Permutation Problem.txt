void solve() {

   for (ll i=1;i<=n;i++){
        verif[i]=0;
   }
   h.pb(1);
   for (ll i=2;i<=n;i++){
    if (verif[i]==0){
        verif[i]=1;
        h.pb(i);
        ll j=i;
        while (2*j<=n && verif[2*j]==0){
            verif[2*j]=1;
            h.pb(2*j);
            j=2*j;
        }
    }
   }
   for (ll i=0;i<h.size();i++){
    cout<<h[i]<<" ";
   }
   cout<<endl;



}
