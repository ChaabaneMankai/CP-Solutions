void dfs(int pos){
    vis[pos]=true;
    for (int i=0;i<graph[pos].size();i++){
        int x=graph[pos][i].first;
        if (!vis[x]){
            ll e=graph[pos][i].second+val[pos];
            val[x]=e;
            dfs(x);
        }
        else if(val[x]!=graph[pos][i].second+val[pos]){
            v=false;
            return;
        }
    }
    return;
}
