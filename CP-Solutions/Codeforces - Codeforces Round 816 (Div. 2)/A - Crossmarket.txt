ll n,q;
string tree[4*N];
string a[N];
string mrg(string x,string y){
    //modifier cette fonction seulement
    string z=x+y;
    return z;
}

void build(int id=0,int ns=0,int ne=n-1){
    if (ns==ne){
        tree[id]=a[ns];
        return ;
    }
    int l=2*id+1;
    int r=l+1;
    int md=ns+(ne-ns)/2;
    build(l,ns,md);
    build(r,md+1,ne);
    tree[id]=mrg(tree[l],tree[r]);
}
string query(int qs,int qe,int id=0,int ns=0,int ne=n-1){
    //cout<<"id"<<id<<endl;
    if (ns>qe || qs>ne)
        return ""; //infinity
    if (qs<=ns && qe>=ne){
    //    cout<<"id"<<id<<endl;
        return tree[id];
    }
    int l=2*id+1;
    int r=l+1;
    int md=ns+(ne-ns)/2;
    return mrg(query(qs,qe,l,ns,md),query(qs,qe,r,md+1,ne));
}



//changer une valeur
void upd(int pos,string val,int id=0,int ns=0,int ne=n-1){
    if (ns>pos || pos>ne)
        return ;
    if (ns==ne){
        tree[id]=val;
        return ;
    }
    int l=2*id+1;
    int r=l+1;
    int md=ns+(ne-ns)/2;
    upd(pos,val,l,ns,md);
    upd(pos,val,r,md+1,ne);
    tree[id]=mrg(tree[l],tree[r]);
}
